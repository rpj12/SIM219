def texttolist(file, expiration=False):
    f = open(file, "r")
    file = []
    for elt in f.readlines():
        [a,b]=elt.split(':', 1)
        if expiration :
            file.append([a,b[:-41]])
        else :
            file.append([a,b[:-1]])
    data = pd.DataFrame(file)
    data.columns = ['login','password']
    return data
